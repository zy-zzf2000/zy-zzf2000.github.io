一个基于动态调整大小数组的队列（FIFO）实现

2019-09-18

在java中，队列的实现由两种基本方式，一种是基于数组的实现方式，另一种是基于链表的实现方式。本文介绍基于数组的一种实现方式。

package zy;

public class Myqueue<Item> {
	private Item[] a=(Item[]) new Object[2];
	private int N=0,head=0,tail=0;
	
	public Myqueue() {
	}
	
	public void resize(int newsize) {  //给数组重新分配空间
		Item[] array=(Item[]) new Object[newsize];
		for(int i=0;i<N;i++) {
			array[i]=a[head+i];
		}
		a=array;         //每次重新分配空间都要重置头索引和尾索引
		head=0;
		N-=head;
		tail=N-1+head;
	}
	
	public boolean isEmpty() {  
		return N==0;
	}
	
	public void enqueue(Item item) {  //入队操作
		if(isEmpty()) {
			a[tail]=item;
			N++;
		}
		else {	
			if(N==a.length) {
			resize(a.length*2);
			}
			int loca =tail+1;
			a[loca]=item;
			N++;
			tail++;
	    }		
	}
	
	public Item dequeue() {  //出队操作
		if(N==a.length/4) {
			resize(a.length/2);
		}
		Item item=a[head];
		a[head]=null;
		head++;
		N--;
		return item;
	}
	
	public static void main(String[] args) {
		Myqueue<String> queue=new Myqueue<String>();
		queue.enqueue("H");
		queue.enqueue("e");
		queue.enqueue("l");
		queue.enqueue("l");
		queue.enqueue("o");
		queue.enqueue(",");
		queue.enqueue("W");
		queue.enqueue("o");
		queue.enqueue("r");
		queue.enqueue("l");
		queue.enqueue("d");
		queue.enqueue("!");
		for(int j=0;j<12;j++) {
			System.out.print(queue.dequeue());
		}
	}		
}
